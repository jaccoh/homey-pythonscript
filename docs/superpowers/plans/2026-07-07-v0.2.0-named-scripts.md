```python
# pythonscript/script_manager.py
from pathlib import Path


class ScriptManager:
    def __init__(self, scripts_root: Path):
        self._root = Path(scripts_root)

    def script_path(self, name: str) -> Path:
        return self._root / f"{name}.py"

    def list_scripts(self) -> list[dict]:
        if not self._root.exists():
            return []
        return [
            {"name": p.stem, "size": p.stat().st_size}
            for p in sorted(self._root.glob("*.py"))
        ]

    def get_script(self, name: str) -> str:
        path = self.script_path(name)
        if not path.exists():
            raise FileNotFoundError(f"Script not found: {name}")
        return path.read_text()

    def save_script(self, name: str, code: str) -> None:
        self._root.mkdir(parents=True, exist_ok=True)
        self.script_path(name).write_text(code)

    def delete_script(self, name: str) -> None:
        path = self.script_path(name)
        if path.exists():
            path.unlink()
```

```python
# api.py
import os
import re
from pathlib import Path

from pythonscript.venv_manager import VenvManager
from pythonscript.script_manager import ScriptManager

_VENV_ROOT = Path(os.environ.get("VENV_ROOT", "/userdata/venvs"))
_SCRIPTS_ROOT = Path(os.environ.get("SCRIPTS_ROOT", "/userdata/scripts"))
_NAME_RE = re.compile(r'^[\w\-]+$')


def _sm() -> ScriptManager:
    return ScriptManager(scripts_root=_SCRIPTS_ROOT)


async def venvs(homey, **kwargs):
    vm = VenvManager(venv_root=_VENV_ROOT)
    return vm.list_venvs()


async def delete_venv(homey, **kwargs):
    uid = str(kwargs.get('uid') or (kwargs.get('body') or {}).get('uid') or '')
    if not uid or not _NAME_RE.match(uid):
        raise ValueError(f"Missing or invalid uid: {uid!r}")
    VenvManager(venv_root=_VENV_ROOT).delete(uid)
    return None


async def scripts(homey, **kwargs):
    return _sm().list_scripts()


async def get_script(homey, **kwargs):
    name = str((kwargs.get('body') or {}).get('name') or kwargs.get('name') or '')
    if not name or not _NAME_RE.match(name):
        raise ValueError(f"Invalid script name: {name!r}")
    code = _sm().get_script(name)
    return {"name": name, "code": code}


async def save_script(homey, **kwargs):
    body = kwargs.get('body') or {}
    name = str(body.get('name') or kwargs.get('name') or '')
    code = str(body.get('code') or kwargs.get('code') or '')
    if not name or not _NAME_RE.match(name):
        raise ValueError(f"Invalid script name: {name!r}")
    _sm().save_script(name, code)
    return None


async def delete_script(homey, **kwargs):
    body = kwargs.get('body') or {}
    name = str(body.get('name') or kwargs.get('name') or '')
    if not name or not _NAME_RE.match(name):
        raise ValueError(f"Invalid script name: {name!r}")
    _sm().delete_script(name)
    return None
```

```python
async def run_script_api(homey, **kwargs):
    """Run a named script from the settings page IDE."""
    import asyncio
    from pythonscript.executor import Executor
    from pythonscript.venv_manager import VenvManager

    body = kwargs.get('body') or {}
    name = str(body.get('name') or kwargs.get('name') or '')
    args = body.get('args') or kwargs.get('args') or None
    timeout = int(body.get('timeout') or kwargs.get('timeout') or 30)

    if not name or not _NAME_RE.match(name):
        raise ValueError(f"Invalid script name: {name!r}")

    code = _sm().get_script(name)
    executor = Executor(sdk=homey, venv_root=_VENV_ROOT)
    result = await executor.run(
        script=code,
        args=args,
        sandbox=True,
        requirements="",
        timeout=timeout,
        card_uid="sandboxed",
    )
    return result.homey_tokens
```

```python
async def test_concurrent_builds_dont_corrupt(tmp_venv_dir):
    """Two concurrent builds of the same uid should serialize, not interleave."""
    vm = VenvManager(venv_root=tmp_venv_dir)
    call_order = []

    original_build = vm.build
    async def tracked_build(card_uid, requirements):
        call_order.append(f"start:{card_uid}")
        await asyncio.sleep(0.01)
        await original_build(card_uid, requirements)
        call_order.append(f"end:{card_uid}")

    vm.build = tracked_build

    # Two concurrent builds of same uid
    await asyncio.gather(
        vm.build("test-uid", "requests==2.31.0"),
        vm.build("test-uid", "requests==2.31.0"),
    )
    # With lock: start,end,start,end (serialized). Without: start,start,end,end
    starts = [i for i, x in enumerate(call_order) if x.startswith("start")]
    ends = [i for i, x in enumerate(call_order) if x.startswith("end")]
```
